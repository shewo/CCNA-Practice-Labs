        // Initialize EmailJS
        (function(){
            emailjs.init("wtFvwtBG_-EvVGtXz"); // Replace with your EmailJS public key
        })();

        // Form submission handler with EmailJS
        document.getElementById('contactForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const submitBtn = document.getElementById('submitBtn');
            const messageArea = document.getElementById('messageArea');
            
            // Show loading state
            submitBtn.innerHTML = '<i class="fas fa-spinner fa-spin me-2"></i>Transmitting...';
            submitBtn.disabled = true;
            
            // Clear previous messages
            messageArea.innerHTML = '';
            
            // Get current date and time
            const now = new Date();
            const submissionTime = now.toISOString().replace('T', ' ').substring(0, 19);
            
            // Prepare template params
            const templateParams = {
                from_name: this.firstName.value + ' ' + this.lastName.value,
                from_email: this.email.value,
                company: this.company.value,
                phone: this.phone.value,
                country: this.country.options[this.country.selectedIndex].text,
                role: this.role.options[this.role.selectedIndex].text,
                message: this.message.value,
                submission_time: submissionTime,
                user_agent: navigator.userAgent
            };
            
            // Send email using EmailJS
            emailjs.send('service_wqrs1ep', 'template_ucpovw9', templateParams) // Replace with your service and template IDs
                .then(function(response) {
                    console.log('SUCCESS!', response.status, response.text);
                    
                    // Show success message
                    messageArea.innerHTML = `
                        <div class="alert alert-success">
                            <i class="fas fa-check-circle me-2"></i>
                            Message successfully transmitted! We'll contact you soon.
                        </div>
                    `;
                    
                    // Reset form
                    document.getElementById('contactForm').reset();
                    
                }, function(error) {
                    console.log('FAILED...', error);
                    
                    // Show error message
                    messageArea.innerHTML = `
                        <div class="alert alert-danger">
                            <i class="fas fa-exclamation-triangle me-2"></i>
                            Transmission failed. Please try again or contact us directly.
                        </div>
                    `;
                })
                .finally(function() {
                    // Reset button state
                    submitBtn.innerHTML = '<i class="fas fa-paper-plane me-2"></i>Launch Message';
                    submitBtn.disabled = false;
                });
        });
        
        // Add animation on scroll
        const observerOptions = {
            threshold: 0.1,
            rootMargin: '0px 0px -50px 0px'
        };
        
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.style.opacity = '1';
                    entry.target.style.transform = 'translateY(0)';
                }
            });
        }, observerOptions);
